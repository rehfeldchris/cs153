NON-TERMINALS

Error ::= <ERROR>
LOLCodeProgram ::= programStart codeBlock programEnd
codeBlock ::= ( ( expressionOrStatement | Error ) )*
programStart ::= <HAI> <IDENTIFIER>
programEnd ::= <KTHXBYE> <EOF>
expressionOrStatement ::= statement
| expression
statement ::= ifStatement
| whileStatement
| variableAssignmentStatement
| switchStatement
| functionDeclarationStatement
| variableDeclarationStatement
| visibleStatement
| gimmehStatement
| returnStatement
returnStatement ::= <FOUND_YR> expression
visibleStatement ::= <VISIBLE> expression
gimmehStatement ::= <GIMMEH> expression
ifStatement ::= ( <COMMA> )? <O> <RLYQ> <YA> <RLY> ( codeBlock ) ( <NO> <WAI> ( codeBlock ) )? <OIC>
whileStatement ::= <IM_IN_YR> loopInitialization codeBlock <IM_OUTTA_YR> voidIdentifier
loopInitialization ::= voidIdentifier ( unaryExpression ( loopCondition )? )?
loopCondition ::= ( <TIL> booleanExpression | <WILE> booleanExpression )
switchStatement ::= <WTF> ( caseStatement )+ ( defaultStatement )? <OIC>
caseStatement ::= <OMG> constantExpression codeBlock ( gtfo )?
defaultStatement ::= <OMGWTF> codeBlock ( gtfo )?
gtfo ::= <GTFO>
functionDeclarationStatement ::= <HOW_DUZ_I> ( voidIdentifier ) ( functionArguments )? ( codeBlock ) <IF_U_SAY_SO>
functionArguments ::= <YR> voidIdentifier ( <AN> <YR> voidIdentifier )*
functionCallExpression ::= <I_CAN_HAS> <IDENTIFIER> ( expression )+
expression ::= booleanExpression
| simpleExpression
| basicMathExpression
| stringConcatenationExpression
| typeCastExpression
| unaryExpression
| functionCallExpression
basicMathExpression ::= additionExpression
| subtractionExpression
| multiplicationExpression
| divisionExpression
| moduloExpression
| maximumExpression
| minimumExpression
booleanExpression ::= logicalAndExpression
| logicalOrExpression
| logicalXorExpression
| logicalNotExpression
| logicalAndListExpression
| logicalOrListExpression
| equalsExpression
| notEqualsExpression
| greaterThanExpression
| lessThanExpression
constantExpression ::= stringLiteral

stringLiteral ::= <STRING>
characterConstant ::= <CHARACTER>
integerConstant ::= <DECIMAL_INTEGER>
realConstant ::= <REAL_NUMBER>

voidIdentifier ::= <IDENTIFIER>
simpleExpression ::= identifier

binaryOperands ::= expression ( <AN> )? expression
variableArityOperands ::= expression ( <MKAY> | ( <AN> )? variableArityOperands ) 
/*Some basic math operators follow, like addition, subtraction etc...*/
additionExpression ::= <ADD> <OF> binaryOperands
subtractionExpression ::= <SUBTRACT> <OF> binaryOperands
multiplicationExpression ::= <MULTIPLY> <OF> binaryOperands
divisionExpression ::= <DIVIDE> <OF> binaryOperands
moduloExpression ::= <MODULO> <OF> binaryOperands
maximumExpression ::= <BIGGR> <OF> binaryOperands
minimumExpression ::= <SMALLR> <OF> binaryOperands
unaryExpression ::= ( increment | decrement )
decrement ::= <DECREMENT> ( <YR> )? expression
increment ::= <INCREMENT> ( <YR> )? expression 
/* Some statements for assigning and declaring variables follow */
variableDeclarationStatement ::= <I_HAS_A> voidIdentifier ( <ITZ> expression )?
variableAssignmentStatement ::= voidIdentifier <ASSIGNMENT> expression 
/* Some boolean operators follow */
logicalAndExpression ::= <LOGICAL_AND> <OF> binaryOperands
logicalOrExpression ::= <LOGICAL_OR> <OF> binaryOperands
logicalXorExpression ::= <XOR> <OF> binaryOperands
logicalNotExpression ::= <LOGICAL_NOT> expression 
/*"all of foo an bar an baz" in lolcode is like "foo && bar && baz" in java*/
logicalAndListExpression ::= <ALL> <OF> variableArityOperands 
/*"any of foo an bar an baz" in lolcode is like "foo || bar || baz" in java*/
logicalOrListExpression ::= <ANY> <OF> variableArityOperands 
/*comparison operators*/
equalsExpression ::= <LOGICAL_AND> <EQUAL> binaryOperands
notEqualsExpression ::= <NOT_EQUAL> binaryOperands
lessThanExpression ::= simpleExpression <SMALLR> <THAN> expression
greaterThanExpression ::= simpleExpression <BIGGR> <THAN> expression 
/*string concatenation*/
stringConcatenationExpression ::= <SMOOSH> variableArityOperands
